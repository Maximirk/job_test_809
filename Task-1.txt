Задача: Каким будет результат выполнения следующего куска кода:

$a = "0";

/*ОТВЕТ: Выведет 0, потому что isset проверяет значение переменной и возвращает true, если значение отлично от null*/
if (isset($a)) {
    echo 0;
} else {
    echo 1;
}

/*ОТВЕТ: Выведет 0, empty вернет true, потому что перед проверкой значение переменной преобразуется в boolean тип, а 0 это false */
if (empty($a)) {
    echo 0;
} else {
    echo 1;
}

/*ОТВЕТ: Выведет 1, так как значение переменной преобразуется в boolean тип, а 0 это false. */
if ($a) {
    echo 0;
} else {
    echo 1;
}
